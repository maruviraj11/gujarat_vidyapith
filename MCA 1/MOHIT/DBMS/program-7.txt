1. Display the employee name, employee department name, boss name, and boss 
   department name. 

>> SELECT 
    e.Emp_name       AS Employee_Name,
    d.Dept_name      AS Employee_Department,
    boss_emp.Emp_name       AS Boss_Name,
    boss_dept.Dept_name     AS Boss_Department						//THIS PRINTS BOSS RECORD WITH EMPLOYEE RECORDS

	FROM emp e
	JOIN dept d ON e.Emp_Dept_Id = d.Dept_Id
	LEFT JOIN emp boss_emp ON e.Emp_boss_id = boss_emp.Emp_Id
	LEFT JOIN dept boss_dept ON b.Emp_Dept_Id = boss_dept.Dept_Id;

>> SELECT e.Emp_name       AS Employee_Name,
    d.Dept_name      AS Employee_Department,
    boss_emp.Emp_name       AS Boss_Name,
    boss_dept.Dept_name     AS Boss_Department						//THIS PRINTS ONLY EMPLOYEE RECORDS

	FROM emp e
	JOIN dept d ON e.Emp_Dept_Id = d.Dept_Id
	LEFT JOIN emp boss_emp ON e.Emp_boss_id = boss_emp.Emp_Id
	JOIN dept boss_dept ON b.Emp_Dept_Id = boss_dept.Dept_Id;


2. Display the same information as above (employee name, department name, boss 
   name, boss department name) but only for employees whose salary is greater than their 
   boss’s salary. 

>> SELECT 
	e.Emp_name AS Employee_Name, 
	d.Dept_name AS Employee_Department, 
	boss_emp.Emp_name AS Boss_Name, 
	boss_dept.Dept_name AS Boss_Department, 
	e.Emp_salary 

	FROM emp e 
	JOIN dept d ON e.Emp_Dept_Id = d.Dept_Id 
	LEFT JOIN emp boss_emp ON e.Emp_boss_id = boss_emp.Emp_Id 
	JOIN dept boss_dept ON boss_emp.Emp_Dept_Id = boss_dept.Dept_Id 
	WHERE boss_emp.Emp_salary < e.Emp_salary;
